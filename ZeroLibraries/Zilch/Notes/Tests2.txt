class Foobar
{
	function Fib(num : Int32) : Int32
	{
		if (num <= 1)
		{
			return num;
		}

		return Fib(num: num - 1) + Fib(num: num - 2);
	}

	function DoFib() : Int32
	{
		return Fib(num: 36);
	}
}
